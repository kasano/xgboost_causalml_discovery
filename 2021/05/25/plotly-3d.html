import plotly.express as px,\
  plotly.io as pio

pio.renderers.default = 'iframe'


df = px.data.iris()


fig = px.scatter_3d(
  data_frame=df, 
  x='sepal_length', 
  y='sepal_width', 
  z='petal_width',    
  color='petal_length', 
  size='petal_length', 
  size_max=18,       
  symbol='species', 
  opacity=0.7
)

# tight layout
fig.update_layout(margin=dict(l=0, r=0, b=0, t=0))

fig.show()
